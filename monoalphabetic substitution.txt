#include <stdio.h>
#include <string.h>
#include <ctype.h>

void monoalphabetic_cipher(char text[], char key[]);

int main(void) {
    
    char plaintext[100];
    char key[27] = "QWERTYUIOPLKJHGFDSAZXCVBNM";
    printf("Enter the plaintext: ");
    fgets(plaintext, sizeof(plaintext), stdin);

    monoalphabetic_cipher(plaintext, key);

    return 0;
}

void monoalphabetic_cipher(char text[], char key[]) {
    printf("Ciphertext: ");
    
    for (int i = 0, n = strlen(text); i < n; i++) {
        char c = text[i];

        if (isupper(c)) {
         
            printf("%c", key[c - 'A']);
        }
     
        else if (islower(c)) {
           
            printf("%c", tolower(key[c - 'a']));
        }
    
        else {
            printf("%c", c);
        }
    }
    
    printf("\n");
}
