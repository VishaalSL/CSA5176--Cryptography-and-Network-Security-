#include <stdio.h>
#include <string.h>
#include <ctype.h>

void caesar_cipher(char text[], int k);

int main(void) {
    
    char plaintext[100];

    printf("Enter the plaintext: ");
    fgets(plaintext, sizeof(plaintext), stdin);

    int k;
    printf("Enter the shift value (1-25): ");
    scanf("%d", &k);
    
    if (k < 1 || k > 25) {
        printf("Invalid shift value. Please enter a number between 1 and 25.\n");
        return 1;
    }
    caesar_cipher(plaintext, k);

    return 0;
}

void caesar_cipher(char text[], int k) {
    printf("Ciphertext: ");

    for (int i = 0, n = strlen(text); i < n; i++) {
        char c = text[i];
        
        if (isupper(c)) {
            printf("%c", ((c - 'A' + k) % 26) + 'A');
        }
    
        else if (islower(c)) {
            printf("%c", ((c - 'a' + k) % 26) + 'a');
        }

        else {
            printf("%c", c);
        }
    }
    
    printf("\n");
}
